```javascript
// OPTR Path Scoring

];
const recommendation = dcr.evaluatePaths(paths);
// Returns: 'lou-first' (OPTR score: 0.85)

// Promotion Gate: Can we move to outreach?
dcr.governanceGate.addRule('outreach', async (decision) => {
  const hasLegalApproval = decision.approvals.includes('Lou legal review');
  const structuralConcernsResolved = decision.metadata?.concernsAddressed === true;
  return {
    valid: hasLegalApproval && structuralConcernsResolved,
    message: hasLegalApproval ? 'Legal validated' : 'Awaiting Lou review'
  };
});

await dcr.promoteDecision(legalReview.id, 'outreach');
// Promotion blocks until Lou approves + concerns addressed
```
